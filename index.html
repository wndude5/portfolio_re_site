Hero Section: Variable Font 및 Kinetic Typography 애니메이션 // 사용자의 스크롤 속도와 애니메이션을 1:1 매핑 (scrub) gsap.to(".hero-title", { scrollTrigger: { trigger: ".hero-section", start: "top top", end: "bottom bottom", scrub: 1, // 1초 지연을 주어 Framer Motion과 같은 부드러운 감도(Spring 느낌) 연출 }, fontVariationSettings: "'wght' 800", // 폰트 굵기 변화 scale: 1.1, letterSpacing: "-0.02em", ease: "none" }); gsap.to(".hero-subtitle", { scrollTrigger: { trigger: ".hero-section", start: "top top", end: "center center", scrub: true }, opacity: 0, y: -50 }); // 2. Product Detail: 시각적 큐레이션 (패브릭 질감 회전 및 스케일) gsap.to(".texture-img", { scrollTrigger: { trigger: ".product-section", start: "top 80%", end: "bottom 20%", scrub: 1.5, }, rotation: 15, scale: 1.2, transformOrigin: "center center", ease: "power1.inOut" }); // 3. Bento Grid: 마우스 호버 마이크로 인터랙션 (Event Delegation 적용) // 수십 개의 아이템이 있어도 리스너는 1개만 동작하여 INP 및 메모리 최적화 const bentoContainer = document.getElementById('bento-container'); bentoContainer.addEventListener('mousemove', (e) => { const item = e.target.closest('.bento-item'); if (!item) return; // 아이템 내 마우스 위치 계산 const rect = item.getBoundingClientRect(); const x = e.clientX - rect.left; const y = e.clientY - rect.top; // 마우스 커서를 따라다니는 은은한 하이라이트 효과 const bg = item.querySelector('.bento-bg'); if (bg) { gsap.to(bg, { x: x - rect.width / 2, y: y - rect.height / 2, opacity: 1, duration: 0.4, ease: "power2.out" }); } // 아이템 자체의 미세한 틸트(Tilt) 효과 const xPct = (x / rect.width - 0.5) * 2; const yPct = (y / rect.height - 0.5) * 2; gsap.to(item, { rotationY: xPct * 4, rotationX: -yPct * 4, transformPerspective: 500, duration: 0.5, ease: "power2.out" }); }); bentoContainer.addEventListener('mouseout', (e) => { const item = e.target.closest('.bento-item'); if (!item) return; const bg = item.querySelector('.bento-bg'); if (bg) { gsap.to(bg, { opacity: 0, duration: 0.3 }); } // 마우스 아웃 시 원래 상태로 복귀 gsap.to(item, { rotationY: 0, rotationX: 0, duration: 0.7, ease: "elastic.out(1, 0.5)" // 탄성 있는 복귀 }); }); });
